breed [ tortugas tortuga ]

to setup
  reset-ticks
  ;; se crean 10 tortugas
  create-tortugas 10
  ;; cada tortuga da 10 pasos
  ask tortugas [ fd 10 ]
  ;; se cambia el icono de tortuga
  ask tortugas [ set shape "turtle" ]
  tick
end

;; funcion que crea el mundo 1
to crear-mundo-1
  ;; se limpia el mundo
  clear-all
  ;; de todos los patches los ubicados en el cuadrante superior izquierdo deben ser de color verde
  ask patches
    [ ifelse (pxcor <= 0) and (pycor >= 0)
      [ set pcolor green ]
      [ set pcolor blue ]
  ]
end

;; mueve las tortugas que no estan en una zona verde
to mover-y-comprobar
  ask tortugas
    [ ifelse pcolor != green
      [ 
        ;; de no estar en una zona verde se gira entre 0ยบ y 360ยบ
        right random 360
        ;; se dan 10 pasos
        fd 10
      ]
      [ 
        ;; de estar en una zona verde no se da ningun paso
        fd 0
      ]
    ]
  tick
end
